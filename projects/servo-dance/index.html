    • Servo Dance

    This project make a servo motor dance.

    Components Required

    • Arduino Board
    • Servo Motor
    • Power Supply (if needed)
    • Jumper Wires

    Code

    
// Servo Dance for Arduino (Single Servo Version)
#include <Servo.h>

// Create a servo object
Servo servo;

void setup() {
  // Attach the servo to pin 9
  servo.attach(9); // Servo on pin 9
}

void loop() {
  // Smooth wave movement
  for (int angle = 0; angle <= 180; angle += 10) {
    servo.write(angle);
    delay(200);
  }
  for (int angle = 180; angle >= 0; angle -= 10) {
    servo.write(angle);
    delay(200);
  }

  // Fast shake
  for (int i = 0; i < 5; i++) {
    servo.write(60);
    delay(100);
    servo.write(120);
    delay(100);
  }

  // Short vibration around center
  for (int i = 0; i < 10; i++) {
    servo.write(90 + (i % 2 == 0 ? 5 : -5));
    delay(80);
  }

  // Salsa-like sway
  for (int i = 0; i < 4; i++) {
    servo.write(90);
    delay(150);
    servo.write(70);
    delay(150);
    servo.write(110);
    delay(150);
    servo.write(90);
    delay(150);
  }

  // Bow movement
  for (int i = 0; i < 2; i++) {
    servo.write(45);
    delay(500);
    servo.write(90);
    delay(500);
  }

  // Pause before repeating
  delay(500);
}

    How to Use

    1. Connect the servo motor to pin 9 of the Arduino board.
    2. Upload the code to your Arduino board.
    3. Power the Arduino board.
    4. Watch the servo motor dance!
